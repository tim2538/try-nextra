Button
import { useState } from 'react';
import { Button as MUIButton } from '@mui/material';
 
export function Button({ children }) {
  const [count, setCount] = useState(0);
  return (
    <MUIButton
      color="primary"
      variant="outlined"
      onClick={() => setCount(count + 1)}
    >
      {children}
      {count}
    </MUIButton>
  );
}
 
<Button>Click: </Button>; // Render component
